```yaml
name: Generate Snake

on:
  schedule:
    - cron: "0 */24 * * *"
  workflow_dispatch:
  push:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: Platane/snk/svg-only@v3
        with:
          github_user_name: NirajItaliya

            dist/github-snake.svg
            dist/github-snake-dark.svg?palette=github-dark

        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

Run the workflow once manually (`Actions` → `Generate Snake` → `Run workflow`). After ~2 minutes, the snake SVG will appear.
